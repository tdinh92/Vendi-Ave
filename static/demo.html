Assessment Charts JavaScript Library Demo

This demo shows how developers can easily embed property assessment charts in their applications.

How to use in your application:

<!-- Include D3.js and the library -->
<script src="https://d3js.org/d3.v7.min.js"></script>
<script src="/static/assessment-charts.js"></script>

<!-- Create a container for the charts -->
<div id="my-charts-container"></div>

<script>
// Initialize the charts
const charts = new AssessmentCharts({
  apiBaseUrl: 'http://localhost:5000', // Your API URL
  containerId: 'my-charts-container', // Container element ID
  width: 800, // Chart width (optional)
  height: 400 // Chart height (optional)
});

// Load charts for a specific address
charts.loadCharts('4 Fiorenza Drive, Wilmington, MA 01887');
</script>
